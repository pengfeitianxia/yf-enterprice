single或者range属性

			function initDefaultPickerOption(){
				var $pickerOption = $('#default-picker-option');
				$pickerOption.buildPicker({
					"single" : true,		//单个控件
					"apply"  : function(event,picker){
						viewContainerOption.html("apply事件触发  : " +  $pickerOption.startTime());
					},
					"cancel" : function(event,picker){
						viewContainerOption.html("cancel事件触发");
					}
				});
			}
			<input class=" form-control" id="default-picker-option" placeholder="单个控件,带有选择和清除事件">

设置初始值

			function initDefaultPickerOptionSetValue(){
				var $pickerOptionInit = $('#default-picker-option-set-value');
				$pickerOptionInit.buildPicker({
					"apply"  : function(event,picker){					//取值				//取值
						viewContainerOptionInit.html("apply事件触发  : " +  $pickerOptionInit.startTime() + " - " + $pickerOptionInit.endTime()) ;
					},
					"cancel" : function(event,picker){
						viewContainerOptionInit.html("cancel事件触发");
					}
				});
				$pickerOptionInit.startTime("2016-07-21");		//设置值(设置值与取值同一个方法名,类似于$("#ABC").val() $("#ABC").val(XXX))
				$pickerOptionInit.endTime("2016-07-28");		//设置值(设置值与取值同一个方法名,类似于$("#ABC").val() $("#ABC").val(XXX))
			}
			<input class=" form-control" id="default-picker-option-set-value" placeholder="p l a c e h o l d e r">

设置初始值 (带时间)

如有需要单独的(时间)控件,(日期+时间)控价,推荐timepicker,datetimepicker. 
			function initDefaultPickerTime(){
				var $pickerOptionTime = $('#default-picker-time');
				$pickerOptionTime.buildPicker({
					"timePicker" : true,						//启用时间控价
					"apply"  : function(event,picker){					//取值				//取值
						viewContainerTime.html("apply事件触发  : " +  $pickerOptionTime.startTime() + " - " + $pickerOptionTime.endTime()) ;
					},
					"cancel" : function(event,picker){
						viewContainerTime.html("cancel事件触发");
					}
				});
				$pickerOptionTime.startTime("2016-07-21 10:23:00");		//设置值(设置值与取值同一个方法名,类似于$("#ABC").val() $("#ABC").val(XXX))
				$pickerOptionTime.endTime("2016-07-28 15:31:00");		//设置值(设置值与取值同一个方法名,类似于$("#ABC").val() $("#ABC").val(XXX))
			}
			<input class=" form-control" id="default-picker-time" placeholder="带有时间的picker">
